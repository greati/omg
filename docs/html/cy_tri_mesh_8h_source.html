229 inline void TriMesh::ComputeBoundingBox()
230 {
231  if ( nv > 0 ) {
232  boundMin=v[0];
233  boundMax=v[0];
234  for ( unsigned int i=1; i<nv; i++ ) {
235  if ( boundMin.x > v[i].x ) boundMin.x = v[i].x;
236  if ( boundMin.y > v[i].y ) boundMin.y = v[i].y;
237  if ( boundMin.z > v[i].z ) boundMin.z = v[i].z;
238  if ( boundMax.x < v[i].x ) boundMax.x = v[i].x;
239  if ( boundMax.y < v[i].y ) boundMax.y = v[i].y;
240  if ( boundMax.z < v[i].z ) boundMax.z = v[i].z;
241  }
242  } else {
243  boundMin.Set(1,1,1);
244  boundMax.Set(0,0,0);
245  }

248 inline void TriMesh::ComputeNormals(bool clockwise)
249 {
250  SetNumNormals(nv);
251  for ( unsigned int i=0; i<nvn; i++ ) vn[i].Set(0,0,0); // initialize all normals to zero
252  for ( unsigned int i=0; i<nf; i++ ) {
253  Point3f N = (v[f[i].v[1]]-v[f[i].v[0]]) ^ (v[f[i].v[2]]-v[f[i].v[0]]); // face normal (not normalized)
254  if ( clockwise ) N = -N;
255  vn[f[i].v[0]] += N;
256  vn[f[i].v[1]] += N;
257  vn[f[i].v[2]] += N;
258  fn[i] = f[i];
259  }
260  for ( unsigned int i=0; i<nvn; i++ ) vn[i].Normalize();
261 }

280  char c = fgetc(fp);
281  while ( !feof(fp) ) {
282  while ( isspace(c) && ( !feof(fp) || c!='\0' ) ) c = fgetc(fp); // skip empty space
283  if ( c == '#' ) while ( !feof(fp) && c!='\n' && c!='\r' && c!='\0' ) c = fgetc(fp); // skip comment line
284  else break;
285  }
286  int i=0;
287  bool inspace = false;
288  while ( i<1024-1 ) {
289  if ( feof(fp) || c=='\n' || c=='\r' || c=='\0' ) break;
290  if ( isspace(c) ) { // only use a single space as the space character
291  inspace = true;
292  } else {
293  if ( inspace ) data[i++] = ' ';
294  inspace = false;
295  data[i++] = c;
296  }
297  c = fgetc(fp);
298  }
299  data[i] = '\0';
300  readLine = i;
301  return i;
302  }
