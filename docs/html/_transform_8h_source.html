194  static Transform make_rotate(float theta, const Vec3& axis) {
195  auto ax = tao::unitize(axis);
196  float sin_theta = std::sin(tao::geometry::radians(theta));
197  float cos_theta = std::cos(tao::geometry::radians(theta));
198  Matrix4x4 mat = Matrix4x4::identity();
199  // first basis vector
200  mat(0, 0) = ax(0) * ax(0) + (1 - ax(0) * ax(0)) * cos_theta;
201  mat(0, 1) = ax(0) * ax(1) * (1 - cos_theta) - ax(2) * sin_theta;
202  mat(0, 2) = ax(0) * ax(2) * (1 - cos_theta) + ax(1) * sin_theta;
203  mat(0, 3) = 0;
204  // second basis vector
205  mat(1, 0) = ax(0) * ax(1) * (1 - cos_theta) + ax(2) * sin_theta;
206  mat(1, 1) = ax(1) * ax(1) + (1 - ax(1) * ax(1)) * cos_theta;
207  mat(1, 2) = ax(1) * ax(2) * (1 - cos_theta) - ax(0) * sin_theta;
208  mat(1, 3) = 0;
209  // third basis vector
210  mat(2, 0) = ax(0) * ax(2) * (1 - cos_theta) - ax(1) * sin_theta;
211  mat(2, 1) = ax(1) * ax(2) * (1 - cos_theta) + ax(0) * sin_theta;
212  mat(2, 2) = ax(2) * ax(2) + (1 - ax(2) * ax(2)) * cos_theta;
213  mat(2, 3) = 0;
214  return Transform(mat, mat.t());
215  }

223  Point3 t_point3(const Point3& p) const {
224  RealValue x = p(0), y = p(1), z = p(2);
225  RealValue xp = mat(0, 0) * x + mat(0, 1) * y + mat(0, 2) * z + mat(0, 3);
226  RealValue yp = mat(1, 0) * x + mat(1, 1) * y + mat(1, 2) * z + mat(1, 3);
227  RealValue zp = mat(2, 0) * x + mat(2, 1) * y + mat(2, 2) * z + mat(2, 3);
228  RealValue wp = mat(3, 0) * x + mat(3, 1) * y + mat(3, 2) * z + mat(3, 3);
229  if (wp == 1) return Point3 {xp, yp, zp};
230  else return Point3 {xp, yp, zp} / wp;
231  }
