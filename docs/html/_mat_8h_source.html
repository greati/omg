482 template<typename T, int M, int N, int P>
483 void multiply(const Mat<T, M, N>& m1, const Mat<T, N, P>& m2,
484  Mat<T, M, P>& m3) {
485  for (auto i = 0; i < m1.nrows(); ++i) {
486  for (auto j = 0; j < m2.ncols(); ++j) {
487  T vm3 = m3(i, j);
488  for (auto k = 0; k < m2.nrows(); ++k) {
489  auto vm1 = m1(i, k);
490  auto vm2 = m2(k, j);
491  m3(i, j) += (vm1 * vm2);
492  }
493  m3(i, j) -= vm3;
494  }
495  }
